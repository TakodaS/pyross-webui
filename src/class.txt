//File myRec.ts
class myRec {
  height: number;
  width: number;

  constructor(height:number, width:number) {
    this.height = height;
    this.width = width;
  }
  // Getter
  get area() {
    return this.calcArea();
  }
  // Method
  calcArea() {
    return this.height * this.width;
  }
}

// Export the named class directly
export { myRec };

//export type myRec = MyFoo;
